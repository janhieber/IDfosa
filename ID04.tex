\subsection*{DCT - Hin}
$\displaystyle F(u) = \frac{C(u)}{\sqrt{N}}
	\cdot \sum_{x=0}^{N-1} \bigg[ f(x)
	\cdot \cos{\frac{(2x+1) \cdot u \cdot \pi{}}{2N}}
	\bigg] \mbox{  für } u=0,1,...,N-1 $
	
$\displaystyle C(u) = \begin{cases}1&\text{für }
	u=0\\\sqrt{2}&\text{sonst}\end{cases} $
